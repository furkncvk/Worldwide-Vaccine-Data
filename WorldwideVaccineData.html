from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.svm import SVR
from sklearn.metrics import mean_squared_error, r2_score

df = pd.read_csv('Worldwide Vaccine Data.csv')

# Hedef deikeni ve bamsz deikenleri belirleme
X = df.drop(['Country', '% of population fully vaccinated'], axis=1)
y = df['% of population fully vaccinated']

# Algoritmalar seme
algorithms = [RandomForestRegressor(), SVR()]

# Metrikleri kaydetmek iin bo listeler oluturma
mse_list = []
r2_list = []

# Her algoritma iin dng
for algorithm in algorithms:
    for _ in range(10):
        # Veriyi TrainSet ve TestSet olarak ayrma
        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=None)

        # Modeli eitme
        model = algorithm.fit(X_train, y_train)

        # Tahmin yapma
        y_pred = model.predict(X_test)

        # Metrikleri hesaplama ve listelere ekleme
        mse_list.append(mean_squared_error(y_test, y_pred))
        r2_list.append(r2_score(y_test, y_pred))

# Her algoritmann ortalama performansn yazdrma
for i, algorithm in enumerate(algorithms):
    print(f"\nAlgorithm {i + 1}: {algorithm.__class__.__name__}")
    print(f"Mean Mean Squared Error (MSE): {sum(mse_list[i*10:(i+1)*10])/10:.4f}")
    print(f"Mean R-squared (R2) Score: {sum(r2_list[i*10:(i+1)*10])/10:.4f}")

# rnek veriler (Bu verileri kendi gerek verilerinizle deitirmeniz gerekebilir)

sensitivity_A = np.array([0.88, 0.90, 0.92, 0.85, 0.87, 0.91, 0.89, 0.88, 0.90, 0.86])
sensitivity_B = np.array([0.82, 0.81, 0.85, 0.84, 0.83, 0.80, 0.82, 0.86, 0.81, 0.84])

specificity_A = np.array([0.75, 0.78, 0.74, 0.77, 0.76, 0.79, 0.75, 0.76, 0.78, 0.80])
specificity_B = np.array([0.72, 0.70, 0.73, 0.75, 0.76, 0.74, 0.77, 0.73, 0.72, 0.75])

# Her bir metrik iin T-Test gerekletiren fonksiyon
def perform_ttest(metric_A, metric_B, metric_name):
    # Eksik deerleri temizle
    metric_A = metric_A[~np.isnan(metric_A)]
    metric_B = metric_B[~np.isnan(metric_B)]

    # T-Test gerekletirme
    t_stat, p_value = ttest_rel(metric_A, metric_B)

    # T-Test sonular
    print(f"T-Test for {metric_name}:")
    print(f"T-statistic: {t_stat:.4f}")
    print(f"P-value: {p_value:.4f}")

    # P-value deerine gre karar verme
    if p_value < 0.05:
        print(f"Statistically significant difference: Algorithm A is better than Algorithm B for {metric_name}.")
    else:
        print(f"No statistically significant difference: Algorithm A and Algorithm B have similar performance for {metric_name}.")
    print("\n")

# Her bir metrik iin T-Test gerekletirme
perform_ttest(accuracy_A, accuracy_B, "Accuracy")
perform_ttest(f1_A, f1_B, "F1 Score")
perform_ttest(sensitivity_A, sensitivity_B, "Sensitivity")
perform_ttest(specificity_A, specificity_B, "Specificity")
